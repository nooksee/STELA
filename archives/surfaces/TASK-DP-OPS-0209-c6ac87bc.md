### 3.4.4 Patch / Diff
.github/hooks/pre-commit — create (mode 755):

    #!/usr/bin/env bash
    set -euo pipefail

    hook_fail() {
      echo "ERROR [pre-commit]: $*" >&2
      exit 1
    }

    repo_root="$(git rev-parse --show-toplevel 2>/dev/null)" || hook_fail "unable to resolve repository root"
    cd "${repo_root}"

    branch="$(git rev-parse --abbrev-ref HEAD)"

    if [[ "$branch" == "main" ]]; then
      hook_fail "direct commits to main are forbidden (PoT §6.2.3). Work on a work/* branch."
    fi

    if [[ "$branch" != "work/"* ]]; then
      hook_fail "commits must be on a work/* branch (PoT §6.2.3). Current branch: ${branch}"
    fi

    bash "${BASH_SOURCE[0]%/*}/llms"

.github/hooks/pre-push — create (mode 755):

    #!/usr/bin/env bash
    set -euo pipefail

    hook_fail() {
      echo "ERROR [pre-push]: $*" >&2
      exit 1
    }

    while IFS=' ' read -r _local_ref _local_sha remote_ref _remote_sha; do
      if [[ "$remote_ref" == "refs/heads/main" ]]; then
        hook_fail "direct push to main is forbidden (PoT §6.1, §6.2.3). Open a PR from a work/* branch."
      fi
    done

ops/bin/hooks — create (mode 755):

    #!/usr/bin/env bash
    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
    REPO_ROOT="$(cd "${SCRIPT_DIR}/../.." && pwd)"
    cd "${REPO_ROOT}"

    git config core.hooksPath .github/hooks
    echo "Hooks configured: core.hooksPath = .github/hooks"
    echo ""
    echo "Active hooks:"
